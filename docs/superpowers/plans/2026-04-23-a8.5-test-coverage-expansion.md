```rust
//! ZeroChecksumTest — RFC 9293 §3.1 MUST-2/3.
//! Ported from tcpreq/tests/checksum.py::ZeroChecksumTest.
//!
//! Probe injects a SYN with a zero TCP checksum. Engine MUST reject:
//! no SYN-ACK emitted; tcp.rx_bad_csum counter bumps.

use std::sync::atomic::Ordering;

use crate::{ProbeResult, ProbeStatus, TcpreqHarness};

pub fn zero_checksum() -> ProbeResult {
    let mut h = TcpreqHarness::new();
    let _listen = match h.eng.listen(crate::OUR_IP, 5000) {
        Ok(lh) => lh,
        Err(e) => return fail("listen failed", format!("{e:?}")),
    };

    let pre_bad_csum = h
        .eng
        .counters_ref()
        .tcp
        .rx_bad_csum
        .load(Ordering::Relaxed);

    // Craft a valid SYN with MSS, then zero the TCP checksum field
    // (at tcp-header offset 16-17 — which is at Ethernet(14)+IPv4(20)+16 = 50).
    let mut syn = crate::framing::build_tcp_syn_with_mss(
        crate::PEER_IP,
        40000,
        crate::OUR_IP,
        5000,
        /*iss_peer*/ 0x1000,
        /*mss*/ 1460,
    );
    let tcp_csum_offset = 14 + 20 + 16;
    syn[tcp_csum_offset] = 0;
    syn[tcp_csum_offset + 1] = 0;

    if let Err(e) = h.eng.inject_rx_frame(&syn) {
        return fail("inject failed", format!("{e:?}"));
    }

    // Engine must drop; no SYN-ACK emitted.
    let out = crate::drain_all_tx(&h.eng);
    if !out.is_empty() {
        return fail(
            "engine emitted a frame in response to zero-csum SYN",
            format!(
                "expected no TX; got {} frames, first={:02x?}",
                out.len(),
                &out[0][..std::cmp::min(out[0].len(), 32)]
            ),
        );
    }

    let post_bad_csum = h
        .eng
        .counters_ref()
        .tcp
        .rx_bad_csum
        .load(Ordering::Relaxed);
    if post_bad_csum != pre_bad_csum + 1 {
        return fail(
            "tcp.rx_bad_csum did not bump by 1",
            format!("pre={pre_bad_csum} post={post_bad_csum}"),
        );
    }

    ProbeResult {
        clause_id: "MUST-2/3",
        probe_name: "ZeroChecksum",
        status: ProbeStatus::Pass,
        message: "zero-csum SYN dropped; tcp.rx_bad_csum bumped; no TX response".into(),
    }
}

fn fail(short: &'static str, detail: String) -> ProbeResult {
    ProbeResult {
        clause_id: "MUST-2/3",
        probe_name: "ZeroChecksum",
        status: ProbeStatus::Fail(short.to_string()),
        message: detail,
    }
}
```

```rust
//! OptionSupportTest (MUST-4) + UnknownOptionTest (MUST-6) +
//! IllegalLengthOptionTest (MUST-7). Ported from tcpreq/tests/options.py.

use std::sync::atomic::Ordering;

use crate::{ProbeResult, ProbeStatus, TcpreqHarness};

/// MUST-4: EOL (kind 0), NOP (kind 1), MSS (kind 2) must all be supported
/// in a SYN. Probe crafts a SYN with all three and asserts handshake
/// completes (SYN-ACK emitted, peer_mss stored correctly).
pub fn option_support() -> ProbeResult {
    let mut h = TcpreqHarness::new();
    let listen = match h.eng.listen(crate::OUR_IP, 5000) {
        Ok(lh) => lh,
        Err(e) => return fail("MUST-4", "OptionSupport", "listen failed", format!("{e:?}")),
    };

    // Craft SYN with <MSS, NOP, NOP, EOL> — full support for all three
    // basic option kinds. The base helper emits NOP padding by default.
    let syn = crate::framing::build_tcp_syn_with_mss(
        crate::PEER_IP, 40000, crate::OUR_IP, 5000,
        /*iss_peer*/ 0x1000, /*mss*/ 1460);
    if let Err(e) = h.eng.inject_rx_frame(&syn) {
        return fail("MUST-4", "OptionSupport", "inject failed", format!("{e:?}"));
    }

    let synack = match crate::drain_one_tx(&h.eng) {
        Some(f) => f,
        None => return fail("MUST-4", "OptionSupport", "no SYN-ACK emitted", String::new()),
    };
    if !crate::is_syn_ack(&synack) {
        return fail("MUST-4", "OptionSupport", "TX not a SYN-ACK", String::new());
    }

    // Complete the handshake — if the engine rejected an option, we'd
    // never get here.
    let ack = crate::framing::build_ack_for(&synack);
    if let Err(e) = h.eng.inject_rx_frame(&ack) {
        return fail("MUST-4", "OptionSupport", "final-ACK inject failed", format!("{e:?}"));
    }
    if h.eng.accept_next(listen).is_none() {
        return fail("MUST-4", "OptionSupport", "accept_next returned None", String::new());
    }

    ProbeResult {
        clause_id: "MUST-4",
        probe_name: "OptionSupport",
        status: ProbeStatus::Pass,
        message: "EOL+NOP+MSS all accepted; handshake complete".into(),
    }
}

/// MUST-6: unknown option kinds MUST be ignored (segment still processed).
/// Probe crafts a SYN with MSS + a reserved-kind unknown option; asserts
/// handshake completes normally.
pub fn unknown_option() -> ProbeResult {
    let mut h = TcpreqHarness::new();
    let listen = match h.eng.listen(crate::OUR_IP, 5000) {
        Ok(lh) => lh,
        Err(e) => return fail("MUST-6", "UnknownOption", "listen failed", format!("{e:?}")),
    };

    // Craft SYN with normal MSS option, then append a reserved option
    // kind (253 — RFC 3692-style experiment kind, guaranteed not
    // standardized) with length=4 + 2 bytes of payload. Must be
    // accepted + ignored.
    let mut syn = crate::framing::build_tcp_syn_with_mss(
        crate::PEER_IP, 40000, crate::OUR_IP, 5000, 0x1000, 1460);
    // Insert an unknown option right before the end-of-options. Use
    // the helper if available; otherwise splice bytes manually.
    let tcp_off = 14 + 20;
    // Find the TCP header end (DataOffset * 4), append option there, grow
    // DataOffset by 1 (4 bytes added: kind+len+2-byte payload), recompute
    // TCP csum.
    let do_byte = syn[tcp_off + 12];
    let tcp_hdr_len = ((do_byte >> 4) as usize) * 4;
    let opt_insert = tcp_off + tcp_hdr_len;
    let unknown_opt = [253u8, 4u8, 0xAB, 0xCD];  // kind 253, len 4, 2 bytes payload
    syn.splice(opt_insert..opt_insert, unknown_opt.iter().copied());
    syn[tcp_off + 12] = ((tcp_hdr_len as u8 + 4) / 4) << 4;
    // IP total-length needs to grow by 4.
    let ip_len_off = 14 + 2;
    let ip_len = u16::from_be_bytes([syn[ip_len_off], syn[ip_len_off + 1]]);
    let new_ip_len = ip_len + 4;
    syn[ip_len_off] = (new_ip_len >> 8) as u8;
    syn[ip_len_off + 1] = (new_ip_len & 0xFF) as u8;
    // Recompute IP checksum.
    crate::framing::recompute_ip_csum(&mut syn);
    // Recompute TCP checksum.
    crate::framing::recompute_tcp_csum(&mut syn, tcp_off);

    if let Err(e) = h.eng.inject_rx_frame(&syn) {
        return fail("MUST-6", "UnknownOption", "inject failed", format!("{e:?}"));
    }
    let synack = match crate::drain_one_tx(&h.eng) {
        Some(f) => f,
        None => return fail(
            "MUST-6", "UnknownOption",
            "no SYN-ACK — unknown option caused a drop",
            String::new(),
        ),
    };
    if !crate::is_syn_ack(&synack) {
        return fail("MUST-6", "UnknownOption", "TX not a SYN-ACK", String::new());
    }

    let ack = crate::framing::build_ack_for(&synack);
    if let Err(e) = h.eng.inject_rx_frame(&ack) {
        return fail("MUST-6", "UnknownOption", "final-ACK inject failed", format!("{e:?}"));
    }
    if h.eng.accept_next(listen).is_none() {
        return fail("MUST-6", "UnknownOption", "accept_next returned None", String::new());
    }

    ProbeResult {
        clause_id: "MUST-6",
        probe_name: "UnknownOption",
        status: ProbeStatus::Pass,
        message: "unknown option (kind 253) ignored; handshake complete".into(),
    }
}

/// MUST-7: illegal option length MUST NOT crash. Engine should drop the
/// segment cleanly (tcp.rx_bad_option bumps OR no crash + no SYN-ACK).
pub fn illegal_length() -> ProbeResult {
    let mut h = TcpreqHarness::new();
    let _listen = match h.eng.listen(crate::OUR_IP, 5000) {
        Ok(lh) => lh,
        Err(e) => return fail("MUST-7", "IllegalLength", "listen failed", format!("{e:?}")),
    };

    let pre_bad = h
        .eng
        .counters_ref()
        .tcp
        .rx_bad_option
        .load(Ordering::Relaxed);

    // Craft SYN with a malformed Timestamp option (kind 8) claiming
    // length=0 (illegal — TS is always 10 bytes). Parser must reject.
    let mut syn = crate::framing::build_tcp_syn_with_mss(
        crate::PEER_IP, 40000, crate::OUR_IP, 5000, 0x1000, 1460);
    let tcp_off = 14 + 20;
    let do_byte = syn[tcp_off + 12];
    let tcp_hdr_len = ((do_byte >> 4) as usize) * 4;
    let opt_insert = tcp_off + tcp_hdr_len;
    let bad_ts = [8u8, 0u8];  // kind=TS, length=0 (ILLEGAL)
    syn.splice(opt_insert..opt_insert, bad_ts.iter().copied());
    // Grow the TCP header by 2 bytes — round up to 4-byte multiple by
    // appending 2 NOPs so DataOffset stays integer.
    syn.splice(opt_insert + 2..opt_insert + 2, [1u8, 1u8].iter().copied());
    syn[tcp_off + 12] = ((tcp_hdr_len as u8 + 4) / 4) << 4;
    let ip_len_off = 14 + 2;
    let ip_len = u16::from_be_bytes([syn[ip_len_off], syn[ip_len_off + 1]]);
    let new_ip_len = ip_len + 4;
    syn[ip_len_off] = (new_ip_len >> 8) as u8;
    syn[ip_len_off + 1] = (new_ip_len & 0xFF) as u8;
    crate::framing::recompute_ip_csum(&mut syn);
    crate::framing::recompute_tcp_csum(&mut syn, tcp_off);

    // Engine must not crash on inject. Must return a result (Ok or Err
    // — both are acceptable; the spec is "don't panic").
    let inject_ok = h.eng.inject_rx_frame(&syn).is_ok();

    let post_bad = h
        .eng
        .counters_ref()
        .tcp
        .rx_bad_option
        .load(Ordering::Relaxed);

    let out = crate::drain_all_tx(&h.eng);

    // Accept either: counter bumped (parser rejected) OR no SYN-ACK
    // emitted (segment silently dropped). Both are spec-compliant.
    if post_bad == pre_bad && !out.is_empty() && crate::is_syn_ack(&out[0]) {
        return fail(
            "MUST-7", "IllegalLength",
            "engine accepted malformed-TS SYN and completed handshake",
            format!("inject_ok={inject_ok}; rx_bad_option unchanged"),
        );
    }

    ProbeResult {
        clause_id: "MUST-7",
        probe_name: "IllegalLength",
        status: ProbeStatus::Pass,
        message: format!(
            "illegal TS length handled without crash; rx_bad_option delta={}, TX={} frames",
            post_bad - pre_bad,
            out.len()
        ),
    }
}

fn fail(
    clause_id: &'static str,
    probe_name: &'static str,
    short: &'static str,
    detail: String,
) -> ProbeResult {
    ProbeResult {
        clause_id,
        probe_name,
        status: ProbeStatus::Fail(short.to_string()),
        message: detail,
    }
}
```

```rust
/// Recompute IP checksum for an Ethernet+IPv4 frame. Zero + sum over
/// the 20-byte IP header with one's-complement fold.
pub fn recompute_ip_csum(frame: &mut [u8]) {
    let ip_off = 14;
    frame[ip_off + 10] = 0;
    frame[ip_off + 11] = 0;
    let mut sum: u32 = 0;
    for i in (0..20).step_by(2) {
        sum += ((frame[ip_off + i] as u32) << 8) | (frame[ip_off + i + 1] as u32);
    }
    while sum >> 16 != 0 {
        sum = (sum & 0xFFFF) + (sum >> 16);
    }
    let csum = !(sum as u16);
    frame[ip_off + 10] = (csum >> 8) as u8;
    frame[ip_off + 11] = (csum & 0xFF) as u8;
}
```

```rust
    let _listen = match h.eng.listen(crate::OUR_IP, 5000) {
        Ok(lh) => lh,
        Err(e) => return fail_mss("mss_support listen failed", format!("{e:?}")),
    };

    let syn = crate::framing::build_tcp_syn_with_mss(
        crate::PEER_IP, 40000, crate::OUR_IP, 5000, 0x1000, 1460);
    if let Err(e) = h.eng.inject_rx_frame(&syn) {
        return fail_mss("inject failed", format!("{e:?}"));
    }

    let synack = match crate::drain_one_tx(&h.eng) {
        Some(f) => f,
        None => return fail_mss("no SYN-ACK", String::new()),
    };
    if !crate::is_syn_ack(&synack) {
        return fail_mss("TX not a SYN-ACK", String::new());
    }

    // Inspect TCP option block of the SYN-ACK — must contain kind=2 (MSS)
    // with length=4.
    let tcp_off = 14 + 20;
    let do_byte = synack[tcp_off + 12];
    let tcp_hdr_len = ((do_byte >> 4) as usize) * 4;
    let opts = &synack[tcp_off + 20..tcp_off + tcp_hdr_len];
    let mut found_mss_kind = false;
    let mut i = 0;
    while i < opts.len() {
        match opts[i] {
            0 => break,                // EOL
            1 => { i += 1; continue; } // NOP
            2 => {
                // MSS: length 4, mss 2 bytes
                if i + 4 <= opts.len() && opts[i + 1] == 4 {
                    found_mss_kind = true;
                    break;
                } else {
                    return fail_mss("MSS option present but malformed",
                        format!("bytes={:02x?}", &opts[i..std::cmp::min(i + 4, opts.len())]));
                }
            }
            _ => {
                // skip other options using their length byte
                if i + 1 >= opts.len() {
                    return fail_mss("truncated option", format!("at byte {i}"));
                }
                let olen = opts[i + 1] as usize;
                if olen < 2 || i + olen > opts.len() {
                    return fail_mss("invalid option length", format!("len={olen} at byte {i}"));
                }
                i += olen;
            }
        }
    }

    if !found_mss_kind {
        return fail_mss(
            "SYN-ACK did not carry MSS option",
            format!("opts bytes={:02x?}", opts),
        );
    }

    ProbeResult {
        clause_id: "MUST-14",
        probe_name: "MSSSupport",
        status: ProbeStatus::Pass,
        message: "peer SYN MSS accepted; our SYN-ACK carries MSS option".into(),
    }
}

    ProbeResult {
        clause_id: "MUST-14",
        probe_name: "MSSSupport",
        status: ProbeStatus::Fail(short.to_string()),
        message: detail,
    }
}
```

```rust
//! Reset-Processing: RST flag processed independently of other flags.
//! Ported from tcpreq/tests/rst_ack.py.

use std::sync::atomic::Ordering;

use crate::{ProbeResult, ProbeStatus, TcpreqHarness};

pub fn rst_ack_processing() -> ProbeResult {
    let mut h = TcpreqHarness::new();
    let listen = match h.eng.listen(crate::OUR_IP, 5000) {
        Ok(lh) => lh,
        Err(e) => return fail("listen failed", format!("{e:?}")),
    };

    // --- Scenario A: complete 3WHS, peer sends plain RST, conn closes ---
    let syn = crate::framing::build_tcp_syn_with_mss(
        crate::PEER_IP, 40000, crate::OUR_IP, 5000, 0x1000, 1460);
    if let Err(e) = h.eng.inject_rx_frame(&syn) {
        return fail("A: inject SYN failed", format!("{e:?}"));
    }
    let synack = match crate::drain_one_tx(&h.eng) {
        Some(f) => f,
        None => return fail("A: no SYN-ACK", String::new()),
    };
    let ack = crate::framing::build_ack_for(&synack);
    if let Err(e) = h.eng.inject_rx_frame(&ack) {
        return fail("A: inject final-ACK failed", format!("{e:?}"));
    }
    let _conn = match h.eng.accept_next(listen) {
        Some(c) => c,
        None => return fail("A: accept_next None", String::new()),
    };

    let pre_rst_cnt = h.eng.counters_ref().tcp.rx_rst.load(Ordering::Relaxed);

    // Peer sends plain RST on the ESTABLISHED conn. Compute seq/ack from
    // the SYN-ACK's ack/seq fields.
    let (our_iss_plus_1, peer_next_seq) = crate::parse_seq_ack_from_synack(&synack);
    let rst = build_rst_for_established(
        /*peer*/ crate::PEER_IP, 40000,
        /*us*/ crate::OUR_IP, 5000,
        /*seq*/ peer_next_seq,
        /*ack*/ our_iss_plus_1,
        /*extra_flags*/ 0,
    );
    if let Err(e) = h.eng.inject_rx_frame(&rst) {
        return fail("A: inject RST failed", format!("{e:?}"));
    }
    let post_rst_cnt = h.eng.counters_ref().tcp.rx_rst.load(Ordering::Relaxed);
    if post_rst_cnt != pre_rst_cnt + 1 {
        return fail(
            "A: rx_rst did not bump",
            format!("pre={pre_rst_cnt} post={post_rst_cnt}"),
        );
    }

    // --- Scenario B: RST with extra flags (e.g., RST+URG) still processed ---
    // Start a fresh listen + 3WHS on a different port to isolate state.
    let listen2 = match h.eng.listen(crate::OUR_IP, 5001) {
        Ok(lh) => lh,
        Err(e) => return fail("B: listen failed", format!("{e:?}")),
    };
    let syn2 = crate::framing::build_tcp_syn_with_mss(
        crate::PEER_IP, 40001, crate::OUR_IP, 5001, 0x2000, 1460);
    if let Err(e) = h.eng.inject_rx_frame(&syn2) {
        return fail("B: inject SYN failed", format!("{e:?}"));
    }
    let synack2 = match crate::drain_one_tx(&h.eng) {
        Some(f) => f,
        None => return fail("B: no SYN-ACK", String::new()),
    };
    let ack2 = crate::framing::build_ack_for(&synack2);
    if let Err(e) = h.eng.inject_rx_frame(&ack2) {
        return fail("B: inject final-ACK failed", format!("{e:?}"));
    }
    let _ = h.eng.accept_next(listen2);

    let mid_rst_cnt = h.eng.counters_ref().tcp.rx_rst.load(Ordering::Relaxed);
    let (our_iss_plus_1_b, peer_next_b) = crate::parse_seq_ack_from_synack(&synack2);
    let rst_urg = build_rst_for_established(
        crate::PEER_IP, 40001, crate::OUR_IP, 5001,
        peer_next_b, our_iss_plus_1_b,
        /*extra_flags*/ 0x20,  // URG bit set alongside RST
    );
    if let Err(e) = h.eng.inject_rx_frame(&rst_urg) {
        return fail("B: inject RST+URG failed", format!("{e:?}"));
    }
    let post_rst_urg_cnt = h.eng.counters_ref().tcp.rx_rst.load(Ordering::Relaxed);
    if post_rst_urg_cnt != mid_rst_cnt + 1 {
        return fail(
            "B: rx_rst did not bump for RST+URG",
            format!("mid={mid_rst_cnt} post={post_rst_urg_cnt}"),
        );
    }

    ProbeResult {
        clause_id: "Reset-Processing",
        probe_name: "RstAck",
        status: ProbeStatus::Pass,
        message: "RST in ESTABLISHED bumps rx_rst; RST+URG also processes as RST".into(),
    }
}

fn build_rst_for_established(
    peer_ip: u32, peer_port: u16, our_ip: u32, our_port: u16,
    seq: u32, ack: u32, extra_flags: u8,
) -> Vec<u8> {
    use dpdk_net_core::test_server::test_packet::build_tcp_frame;
    use dpdk_net_core::tcp_options::TcpOpts;
    const TCP_RST: u8 = 0x04;
    const TCP_ACK_BIT: u8 = 0x10;
    let flags = TCP_RST | TCP_ACK_BIT | extra_flags;
    let mut f = build_tcp_frame(
        peer_ip, peer_port, our_ip, our_port,
        seq, ack, flags, /*win*/ 65535,
        TcpOpts::default(), /*payload*/ &[],
    );
    // If we set URG (extra_flags & 0x20), the urgent-pointer field must
    // be present; the build_tcp_frame helper zeroes it by default which
    // is benign.
    let _ = &mut f;
    f
}
```

```rust
/// POSIX shutdown(2) subset: full-close only. Half-close (SHUT_RD /
/// SHUT_WR) returns -EOPNOTSUPP. See spec §4 + §6.4 row
/// AD-A8.5-shutdown-no-half-close for rationale.
pub const DPDK_NET_SHUT_RD: i32 = 0;
pub const DPDK_NET_SHUT_WR: i32 = 1;
pub const DPDK_NET_SHUT_RDWR: i32 = 2;

/// SHUT_RDWR = equivalent to `dpdk_net_close(h, 0)`.
/// SHUT_RD or SHUT_WR = -EOPNOTSUPP (half-close not implemented).
/// any other `how` = -EINVAL.
/// Returns 0 on success; negative errno on failure.
#[no_mangle]
pub extern "C" fn dpdk_net_shutdown(h: dpdk_net_conn_t, how: i32) -> i32 {
    match how {
        DPDK_NET_SHUT_RDWR => dpdk_net_close(h, 0),
        DPDK_NET_SHUT_RD | DPDK_NET_SHUT_WR => -(libc::EOPNOTSUPP as i32),
        _ => -(libc::EINVAL as i32),
    }
}
```
